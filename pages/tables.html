/* http://www.mocky.io/v2/5790a35926000085167ee533
URL generada con datos aleatorios
*/

var app = angular.module("myapp", ["ngTable"]);

app.controller("tableController",tableController);

tableController.$inject = ["NgTableParams","$http"]
function tableController(NgTableParams,$http){
var self = this;
var url = "http://www.mocky.io/v2/5790a35926000085167ee533/?callback=JSON_CALLBACK";
self.tableParams = new NgTableParams({
  count: 5 // Cantidad de registros que se quieren mostrar al inicio
});

self.tableParams.settings({
  counts: [5,10,20] // Cantidad de paginas que se quiere seleccionar en el paginador
});

// En esta seccion se configuran las columnas, si se desean colocar filtros y si se quiere activar la funcionalidad de filtrado
self.cols = [
  { field: "age", title: "Edad", filter: { age: "number"}, sortable: "age", show: true},
  { field: "company", title: "Empresa", filter: { company: "text"}, sortable: "company",show: true},
  { field: "email", title: "Correo", filter: { email: "text"}, sortable:"email",show: true}
];

// A traves de cualquier peticion, no solo jsonp sino post, get o en general debera traer el array de objetos a colocar en la tabla
// y se colocaran en la variable dataset

$http.jsonp(url).then(function(response){
  console.log(response.data);
  self.tableParams.settings(
    {
      dataset: response.data
    }

  );
});
}
